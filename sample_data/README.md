```json
{
  "timestamp": "2025-09-19T14:30:45.123Z",
  "source_ip": "192.168.1.100",
  "destination_ip": "10.0.0.50",
  "source_port": 49152,
  "destination_port": 443,
  "protocol": "TCP",
  "bytes_transferred": 2048,
  "connection_duration_ms": 150,
  "user_agent": "Mozilla/5.0...",
  "response_code": 200,
  "country_code": "US"
}
```
